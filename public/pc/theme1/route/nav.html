路由 URL方法 链接
返回路由实验首页 Route::get('[:lng]/route/index', 'route/index'); url('route/index')
{:url('route/show', ['lng'=>'ttt'])} Route::get('[:lng]/route/show', 'route/show'); url('route/show', ['lng'=>'ttt'])
{:url('route/show', ['lng'=>''])} Route::get('[:lng]/route/show', 'route/show'); url('route/show', ['lng'=>''])
{:url('route/show2', ['lng'=>'en','name'=>'name'])} Route::get('[:lng]/route/:name', 'route/show2'); url('route/show2', ['lng'=>'en','name'=>'name'])
{:url('route/full', ['full'=>'en'])} Route::get('full/:full$', 'route/full'); url('route/full', ['full'=>'en'])
{:url('route/full', ['full'=>'en','name'=>'name'])} 不匹配--Route::get('full/:full$', 'route/full'); url('route/full', ['full'=>'en','name'=>'name'])
{:url('route_mark', ['name'=>'markname'])} Route::get('mark/:name','route/show2')->name('route_mark'); url('route_mark', ['name'=>'markname'])
